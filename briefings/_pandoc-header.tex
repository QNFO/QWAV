% Pandoc LaTeX header -- mimics business-docs.css styling
% Font: Segoe UI (Win system font) with Inter fallback
\usepackage{fontspec}
\IfFontExistsTF{Inter}{
  \setmainfont{Inter}[Ligatures=TeX]
}{
  \setmainfont{Segoe UI}[Ligatures=TeX]
}
\setmonofont{Consolas}[Scale=0.9]

% Page geometry: A4, margins matching CSS @page
\usepackage[a4paper, top=0.75in, bottom=0.75in, left=1.2in, right=0.8in]{geometry}

% Line spacing ~1.6
\usepackage{setspace}
\setstretch{1.55}

% Page numbers at bottom-right
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[R]{\footnotesize\color{gray}Page \thepage}
\renewcommand{\headrulewidth}{0pt}

% Section heading styles approximating CSS
\usepackage{titlesec}
\usepackage{xcolor}
\definecolor{dark}{RGB}{26,26,26}
\definecolor{medgray}{RGB}{51,51,51}
\definecolor{lightborder}{RGB}{204,204,204}

% H2: 15pt, bottom border 1px #ccc
\titleformat{\section}
  {\normalfont\fontsize{15pt}{18pt}\bfseries\color{dark}}
  {\thesection}{1em}{}
  [\vspace{-0.5em}\textcolor{lightborder}{\rule{\textwidth}{0.5pt}}]

% H3: 13pt
\titleformat{\subsection}
  {\normalfont\fontsize{13pt}{16pt}\bfseries\color{medgray}}
  {\thesubsection}{1em}{}

% H4: 12pt
\titleformat{\subsubsection}
  {\normalfont\fontsize{12pt}{14pt}\bfseries\color{medgray!70}}
  {\thesubsubsection}{1em}{}

% Table styling: alternating row colors
\usepackage{colortbl}
\usepackage{longtable}
\definecolor{tablerow}{RGB}{250,250,250}
\definecolor{tablehead}{RGB}{240,240,240}

% Make longtable rows alternate color
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\rowcolors{2}{white}{tablerow}}

% Table header background
% (pandoc uses longtable; we style th via renewenvironment)
\let\oldLThead\endhead

% Paragraph spacing
\usepackage{parskip}
\setlength{\parskip}{0.5em}
\setlength{\parindent}{0pt}

% Hyperlinks
\usepackage{hyperref}
\hypersetup{colorlinks=true, linkcolor=blue, urlcolor=blue}

% Code blocks
\usepackage{fvextra}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  breaklines, fontsize=\small, bgcolor=lightborder!20, frame=single, framesep=4pt
}
